## Executive Summary

This document outlines the product requirements and SEO strategy for rebuilding jindabynelectrician.com.au as a local electrician website targeting Jindabyne and the Snowy Mountains region. The primary opportunity identified is creating dedicated service and location pages to outrank competitors who rely on single consolidated pages.

### Form Handling with SendGrid

All contact and quote request forms should submit via SendGrid API.

**Required SendGrid Setup:**

1. Create SendGrid account at https://sendgrid.com
2. Verify sender domain (jindabynelectrician.com.au)
3. Create API key with Mail Send permissions
4. Set up email templates (optional but recommended)

**Environment Variables for Vercel:**

Add these in Vercel Dashboard > Project > Settings > Environment Variables:

```
SENDGRID_API_KEY=SG.xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx
SENDGRID_FROM_EMAIL=enquiries@jindabynelectrician.com.au
SENDGRID_TO_EMAIL=admin@jindabynelectrician.com.au
NEXT_PUBLIC_SITE_URL=https://jindabynelectrician.com.au
```

**Important:** When copying API keys to Vercel:
- Do not include quotes around values
- Do not include spaces before or after the value
- Do not include line breaks
- Copy the raw key value only

**API Route Example:**

Create `/app/api/contact/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server'
import sgMail from '@sendgrid/mail'

sgMail.setApiKey(process.env.SENDGRID_API_KEY as string)

export async function POST(request: NextRequest) {
  try {
    const body = await request.json()
    
    const { name, email, phone, suburb, service, message } = body

    // Validate required fields
    if (!name || !email || !phone || !message) {
      return NextResponse.json(
        { error: 'Missing required fields' },
        { status: 400 }
      )
    }

    // Email to business
    const businessEmail = {
      to: process.env.SENDGRID_TO_EMAIL,
      from: process.env.SENDGRID_FROM_EMAIL as string,
      replyTo: email,
      subject: `New Enquiry from ${name} - ${service || 'General'}`,
      html: `
        <h2>New Website Enquiry</h2>
        <p><strong>Name:</strong> ${name}</p>
        <p><strong>Email:</strong> ${email}</p>
        <p><strong>Phone:</strong> ${phone}</p>
        <p><strong>Suburb:</strong> ${suburb || 'Not provided'}</p>
        <p><strong>Service:</strong> ${service || 'Not specified'}</p>
        <p><strong>Message:</strong></p>
        <p>${message}</p>
      `,
    }

    // Confirmation email to customer
    const confirmationEmail = {
      to: email,
      from: process.env.SENDGRID_FROM_EMAIL as string,
      subject: 'Thanks for contacting Jindabyne Electrician',
      html: `
        <h2>Thanks for your enquiry, ${name}!</h2>
        <p>We have received your message and will get back to you within 24 hours.</p>
        <p>If your matter is urgent, please call us directly on <strong>0411 326 021</strong>.</p>
        <br>
        <p>Kind regards,</p>
        <p>The Jindabyne Electrician Team</p>
      `,
    }

    await sgMail.send(businessEmail)
    await sgMail.send(confirmationEmail)

    return NextResponse.json({ success: true })
  } catch (error) {
    console.error('SendGrid error:', error)
    return NextResponse.json(
      { error: 'Failed to send message' },
      { status: 500 }
    )
  }
}
```

**Frontend Form Component Example:**

```typescript
'use client'

import { useState } from 'react'

export default function ContactForm() {
  const [status, setStatus] = useState<'idle' | 'loading' | 'success' | 'error'>('idle')

  async function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault()
    setStatus('loading')

    const formData = new FormData(e.currentTarget)
    const data = Object.fromEntries(formData)

    try {
      const response = await fetch('/api/contact', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data),
      })

      if (response.ok) {
        setStatus('success')
      } else {
        setStatus('error')
      }
    } catch {
      setStatus('error')
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <input type="text" name="name" placeholder="Your Name" required />
      <input type="email" name="email" placeholder="Your Email" required />
      <input type="tel" name="phone" placeholder="Your Phone" required />
      <input type="text" name="suburb" placeholder="Your Suburb/Postcode" />
      <select name="service">
        <option value="">Select a Service</option>
        <option value="general">General Electrical</option>
        <option value="solar">Solar Installation</option>
        <option value="battery">Battery Storage</option>
        <option value="starlink">Starlink Installation</option>
        <option value="underfloor">Underfloor Heating</option>
        <option value="switchboard">Switchboard Upgrade</option>
        <option value="smoke-alarms">Smoke Alarms</option>
        <option value="ev-charger">EV Charger</option>
        <option value="emergency">Emergency Electrical</option>
        <option value="commercial">Commercial Services</option>
        <option value="other">Other</option>
      </select>
      <textarea name="message" placeholder="How can we help?" required />
      <button type="submit" disabled={status === 'loading'}>
        {status === 'loading' ? 'Sending...' : 'Send Enquiry'}
      </button>
      {status === 'success' && <p>Thanks! We will be in touch soon.</p>}
      {status === 'error' && <p>Something went wrong. Please call us instead.</p>}
```
